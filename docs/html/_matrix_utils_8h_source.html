43 template <fgm::Matrix T, fgm::Matrix U>
44 void EXPECT_MAT_EQ(const T& expected, const U& actual)
45 {
46 using ValueType = T::value_type;
47
48 static_assert(T::rows == U::rows && T::columns == U::columns,
49 "Matrices must of the same dimension, e.g: 3x3 and 3x3");
50
51 for (std::size_t i = 0; i < T::rows; ++i)
52 for (std::size_t j = 0; j < T::columns; ++j)
53 if constexpr (std::is_same_v<ValueType, double>)
54 ASSERT_DOUBLE_EQ(expected(i, j), static_cast<ValueType>(actual(i, j)));
55 else if constexpr (std::is_same_v<ValueType, float>)
56 ASSERT_FLOAT_EQ(expected(i, j), static_cast<ValueType>(actual(i, j)));
57 else
58 ASSERT_EQ(expected(i, j), static_cast<ValueType>(actual(i, j)));
